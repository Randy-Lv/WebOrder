CREATE TABLE `Order` (
`id` int(11) NOT NULL AUTO_INCREMENT,
`Title` varchar(100),
`Num`   varchar(10),
`Company` varchar(100),
`UserName` varchar(50),
`Mobile` varchar(20),
`Fax`    varchar(20),
`Remark` varchar(200),
`CreateDate` varchar(20),
 PRIMARY KEY (`id`)

)ENGINE=InnoDB AUTO_INCREMENT=10 DEFAULT CHARSET=utf8;;
